# 希沃白板5修改工具 - 依赖列表

# =============== 基础依赖 ===============
# Tkinter是Python标准库，无需额外安装
pygame>=2.0.0  # 用于音频预览功能
Pillow>=8.0.0  # 用于图像相关处理
pyperclip>=1.8.2  # 用于剪贴板操作

# =============== Rin-UI相关依赖 ===============
# 从 https://github.com/RinLit-233-shiroko/Rin-UI/ 项目获取的组件库
PySide6>=6.5.0  # Qt应用开发框架，Rin-UI的基础依赖

# 安装RinUI的方法：
# 1. 直接从PyPI安装（如果已发布）：
pip install RinUI>=0.1.0

# 2. 或从GitHub克隆并安装（推荐最新版本）：
git clone https://github.com/RinLit-233-shiroko/Rin-UI.git
cd Rin-UI
pip install -e .

# 注意：如果RinUI在PyPI上不可用，请使用GitHub安装方法
# 详细信息请访问：https://github.com/RinLit-233-shiroko/Rin-UI/
